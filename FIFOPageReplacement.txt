import java.util.*;

public class FIFOPageReplacement {
    public static void main(String[] args) {

        int[] pages = {2, 3, 2, 1, 5, 2, 4, 5, 3, 2, 5, 2};
        int framesCount = 3;

        Queue<Integer> frames = new LinkedList<>();
        int pageFaults = 0;
        int pageHits = 0;

        System.out.println("Page\tFrames\t\tHit/Fault");

        for (int page : pages) {

            // If page NOT present -> Page Fault
            if (!frames.contains(page)) {
                pageFaults++;

                if (frames.size() == framesCount) {
                    frames.poll(); // remove oldest page
                }

                frames.add(page);
                System.out.println(page + "\t" + frames + "\tFault");
            } 
            else {
                // Page is already inside frames â†’ HIT
                pageHits++;
                System.out.println(page + "\t" + frames + "\tHit");
            }
        }

        System.out.println("\nTotal Page Faults: " + pageFaults);
        System.out.println("Total Page Hits: " + pageHits);
    }
}
