import java.util.*;

class PriorityProcess {
    String pid;
    int arrival, burst, priority;
    int start, finish, waiting, turnaround;
    boolean completed;

    PriorityProcess(String pid, int arrival, int burst, int priority) {
        this.pid = pid;
        this.arrival = arrival;
        this.burst = burst;
        this.priority = priority;
        this.completed = false;
    }
}

public class Priority_NonPreemptive {

    public static void main(String[] args) {

        PriorityProcess[] p = {
                new PriorityProcess("P1", 0, 8, 1),
                new PriorityProcess("P2", 0, 6, 2),
                new PriorityProcess("P3", 2, 1, 3),
                new PriorityProcess("P4", 3, 2, 0)
        };

        int n = p.length;
        int completed = 0;
        int time = 0;

        System.out.println("PID\tAT\tBT\tPR\tCT\tWT\tTAT");

        while (completed < n) {
            PriorityProcess selected = null;

            // Select arrived process with highest priority (lowest number)
            for (PriorityProcess pr : p) {
                if (!pr.completed && pr.arrival <= time) {
                    if (selected == null || pr.priority < selected.priority) {
                        selected = pr;
                    }
                }
            }

            if (selected == null) {   // No process arrived → CPU idle
                time++;
                continue;
            }

            // Execute the selected process (non-preemptive → complete fully)
            selected.start = time;
            selected.finish = time + selected.burst;
            selected.turnaround = selected.finish - selected.arrival;
            selected.waiting = selected.start - selected.arrival;
            selected.completed = true;

            time = selected.finish;
            completed++;

            System.out.println(selected.pid + "\t" +
                    selected.arrival + "\t" +
                    selected.burst + "\t" +
                    selected.priority + "\t" +
                    selected.finish + "\t" +
                    selected.waiting + "\t" +
                    selected.turnaround);
        }
    }
}
